Unidad 1 – Reconocimiento de Funciones

Descripción mejorada. Se desarrolla un programa que, a partir de una función matemática ingresada por el usuario con un máximo de dos variables, identifica dichas variables y cuantifica el número de operaciones presentes. Se consideran sumas, restas, multiplicaciones, divisiones, potencias y multiplicaciones implícitas (como 5x o xy), evitando confundir nombres de funciones (sin, log, sqrt, etc.) con variables. El resultado presenta: variables detectadas, número de variables distintas y total de operaciones encontradas.

Objetivos de aprendizaje

  • Diferenciar símbolos de función y variables.
  • Detectar y contar operadores explícitos e implícitos.
  • Resumir métricas de complejidad de la expresión.

Entradas y salidas

  • Entrada: cadena con la expresión (hasta 2 variables).
  • Salida: listado de variables, conteos por operador y total.

# Analizador de funciones (2 variables) con conteo detallado
# - Evita confundir funciones: sin, cos, tan, log, ln, exp, sqrt, asin, acos, atan, sinh, cosh, tanh, log10, log2, pow
# - Cuenta +, -, *, /, ^ y multiplicaciones implícitas: 5x, xy, (x+1)(y-2), 2(x+y)
# - Muestra desglose de operadores y total
import re
import tkinter as tk
from tkinter import ttk, messagebox

FUNC_KEYWORDS = {"sin","cos","tan","log","ln","exp","sqrt","asin","acos","atan",
                 "sinh","cosh","tanh","log10","log2","pow"}

TOKEN_RE = re.compile(r"([a-zA-Z]+|\d+(?:\.\d+)?|[\+\-\*\/\^\(\)])")

def detectar_variables(expr: str):
    vars_set = set()
    for m in re.finditer(r"[a-zA-Z]+", expr):
        word = m.group(0)
        if word.lower() not in FUNC_KEYWORDS:
            for ch in word:
                vars_set.add(ch)
    return sorted(vars_set)

def contar_operaciones(expr: str):
    s = expr.replace(" ", "")
    counts = {"+":0,"-":0,"*":0,"/":0,"^":0,"impl":0}
    # explícitos (+ - * / ^) con manejo de '-' unario
    for i,c in enumerate(s):
        if c in "+*/^":
            counts[c]+=1
        elif c=="-":
            if i>0 and s[i-1] not in "+-*/^(":
                counts["-"]+=1
    # implícitas con pares de tokens contiguos
    tokens = TOKEN_RE.findall(s)
    def is_num(t): return re.fullmatch(r"\d+(?:\.\d+)?", t) is not None
    def is_word(t): return re.fullmatch(r"[a-zA-Z]+", t) is not None
    def is_var(t): return is_word(t) and t.lower() not in FUNC_KEYWORDS
    for a,b in zip(tokens,tokens[1:]):
        if a=="*" or b=="*": continue
        if (is_num(a) and is_var(b)) or (is_var(a) and is_var(b)) or \
           (a==")" and b=="(") or (a==")" and is_var(b)) or \
           (is_var(a) and b=="(") or (is_num(a) and b=="("):
            counts["impl"]+=1
    total = sum(v for k,v in counts.items() if k!="impl") + counts["impl"]
    return counts,total

def analizar():
    expr = entrada.get()
    if not expr.strip():
        messagebox.showwarning("Error", "Por favor ingresa una función.")
        return
    variables = detectar_variables(expr)
    if len(variables)==0:
        messagebox.showinfo("Resultado", "No se detectaron variables.")
        return
    if len(variables)>2:
        messagebox.showerror("Error", f"Más de 2 variables: {', '.join(variables)}")
        return
    counts, total = contar_operaciones(expr)
    resultado.set(
        "Variables: {vars}\nN.º de variables: {n}\n"
        "Conteo de operadores:\n"
        "  +: {p}   -: {m}   *: {t}   /: {d}   ^: {e}\n"
        "  Multiplicaciones implícitas: {impl}\n"
        "Total de operaciones: {tot}".format(
            vars=", ".join(variables), n=len(variables),
            p=counts['+'], m=counts['-'], t=counts['*'], d=counts['/'], e=counts['^'],
            impl=counts['impl'], tot=total
        )
    )

# UI
root = tk.Tk()
root.title("Analizador de Funciones (2 variables)")
root.geometry("560x360")
root.configure(bg="#0f162c")
style = ttk.Style(); style.theme_use("clam")
style.configure("TButton", font=("Inter", 11), padding=6)
style.configure("TLabel", font=("Inter", 11))
style.configure("TEntry", padding=6)
tk.Label(root, text="Ingresa una función (máx. 2 variables):",
         font=("Inter", 12, "bold"), bg="#0f162c", fg="#e6ecff").pack(pady=10)
entrada = ttk.Entry(root, font=("Consolas", 14), width=44); entrada.pack(pady=6, ipady=4)
ttk.Button(root, text="Analizar", command=analizar).pack(pady=8)

tk.Label(root, textvariable=resultado, justify="left",
         font=("Consolas", 12), bg="#111b2a", fg="#e6ecff",
         bd=1, relief="groove").pack(fill="both", expand=True, padx=18, pady=12)
root.mainloop()

Unidad 1 – Graficar Funciones Lineales (ASCII)

Descripción mejorada. Una función es una relación entre dos conjuntos que asocia a cada entrada x un único valor de salida y, denotado y = f(x). Las funciones lineales se modelan como f(x) = m·x + b, donde m es la pendiente (inclinación de la recta) y b la ordenada al origen (punto de corte con el eje y).

Se implementa un programa en Python —sin librerías gráficas— capaz de reconocer dos expresiones lineales y generar un gráfico ASCII en un plano cartesiano con rango estándar de ±20 unidades. El resultado muestra ejes, origen e intersección aproximada, facilitando la interpretación de pendientes y cortes.

# Graficador ASCII de dos funciones lineales y = m x + b
# Entrada estilo: 2x+1 , -x+3 , 0.5x-4 etc.
def preparar(expr: str) -> str:
    s = expr.replace(" ", "")
    s = s.replace("^","**")
    s = s.replace("+x","+1*x").replace("-x","-1*x")
    if s.startswith("x"): s = "1*" + s
    out = []
    for i,ch in enumerate(s):
        if ch == "x" and i>0 and (s[i-1].isdigit() or s[i-1]==")"):
            out.append("*x")
        else:
            out.append(ch)
    s = "".join(out).replace("**x","*x")
    return s

def eval_y(func, x):
    try:
        return eval(func, {"__builtins__":{}}, {"x":x})
    except Exception:
        return None

def main():
    f1 = input("Función 1 (ej: 2x+1): ")
    f2 = input("Función 2 (ej: -x+3): ")
    f1p, f2p = preparar(f1), preparar(f2)
    xmin, xmax = -20, 20
    ymin, ymax = -12, 12
    print("\nGráfico ASCII (escala 1 unidad por celda):\n")
    for y in range(ymax, ymin-1, -1):
        row = []
        for x in range(xmin, xmax+1):
            y1 = eval_y(f1p, x); y2 = eval_y(f2p, x)
            on1 = (y1 is not None and abs(y - y1) < 0.5)
            on2 = (y2 is not None and abs(y - y2) < 0.5)
            if on1 and on2: row.append("#")
            elif on1:       row.append("*")
            elif on2:       row.append("o")
            elif x==0 and y==0: row.append("+")
            elif x==0: row.append("|")
            elif y==0: row.append("-")
            else: row.append(" ")
        print("".join(row))
    print("\nLeyenda:\n  * = Función 1   o = Función 2   # = Intersección   | = Eje Y   - = Eje X   + = Origen")
    def mb(expr):
        y0 = eval_y(expr, 0); y1 = eval_y(expr, 1)
        if y0 is None or y1 is None: return None,None
        m = y1 - y0; b = y0
        return m,b
    m1,b1 = mb(f1p); m2,b2 = mb(f2p)
    if None in (m1,b1,m2,b2):
        print("\nNo se pudo estimar intersección.")
    elif abs(m1 - m2) < 1e-12:
        print("\nRectas paralelas o coincidentes (m1 ≈ m2).")
    else:
        xi = (b2 - b1) / (m1 - m2); yi = m1*xi + b1
        print(f"\nIntersección aproximada: (x={xi:.3f}, y={yi:.3f})")

if __name__=="__main__":
    main()

Unidad 1 – Método de Regula Falsi

Descripción mejorada. El método de la Regula Falsi (o falsa posición) es un procedimiento por intervalos para hallar raíces reales de f(x)=0. Parte de un tramo [a,b] tal que f(a)·f(b)<0, lo que asegura un cambio de signo y la existencia de al menos una raíz. En cada iteración se traza la secante entre (a,f(a)) y (b,f(b)) y se toma su intersección con el eje x como aproximación; luego se actualiza el extremo que preserva el cambio de signo, manteniendo acotada la solución.

El método combina la robustez del enfoque por bisección (la raíz nunca escapa del intervalo) con una direccionalidad propia de la secante, que suele acelerar el acercamiento a la raíz. En casos de estancamiento por curvatura, la variante Illinois atenúa el peso de un extremo para recuperar progreso. Los criterios de parada habituales son una tolerancia en |f(x)| y/o en la longitud de [a,b], además de un máximo de iteraciones.

Unidad 1 – Método de Newton-Raphson

Descripción mejorada. El método de Newton–Raphson aproxima raíces reales de funciones continuas y derivables mediante la actualización xₙ₊₁ = xₙ − f(xₙ)/f’(xₙ). Se interpreta como la intersección de la recta tangente en xₙ con el eje x. Con un punto inicial adecuado y funciones suaves, su convergencia suele ser cuadrática, es decir, muy rápida.

Aplicación en Python. Primero se grafica f(x) en un intervalo definido por el usuario para identificar visualmente zonas con posibles raíces. Si se confirma, se solicita un valor inicial x₁ y se ejecuta Newton–Raphson con derivada numérica central, controles ante pendientes casi nulas y criterios de parada por tolerancia y por iteraciones máximas. El programa reporta la raíz aproximada y el número de iteraciones, y muestra la traza de los puntos generados.

import numpy as np
import matplotlib.pyplot as plt

SAFE = {
    "sin": np.sin, "cos": np.cos, "tan": np.tan,
    "exp": np.exp, "log": np.log, "sqrt": np.sqrt,
    "pi": np.pi, "e": np.e, "abs": np.abs
}

def parse_func(s):
    s = s.replace("^","**")
    def f(x):
        return eval(s, {"__builtins__":{}}, {**SAFE, "x":x})
    return f

def deriv(f, x, h=1e-6):
    return (f(x+h) - f(x-h)) / (2*h)

if __name__ == "__main__":
    func_str = input("Ingrese f(x) (ej: x**3 - x - 1): ")
    f = parse_func(func_str)
    xmin = float(input("x mínimo: "))
    xmax = float(input("x máximo: "))
    xs = np.linspace(xmin, xmax, 600)
    ys = f(xs)

    plt.figure(figsize=(8,5))
    plt.plot(xs, ys, label=f"f(x) = {func_str}")
    plt.axhline(0, color="black", linestyle="--", linewidth=0.8)
    plt.axvline(0, color="black", linestyle="--", linewidth=0.8)
    plt.title("Vista previa de f(x)")
    plt.xlabel("x"); plt.ylabel("f(x)")
    plt.grid(True, alpha=.25); plt.legend()
    plt.show()

    op = input("¿Aplicar Newton-Raphson? (s/n): ").strip().lower()
    if op == "s":
        x0 = float(input("Ingrese x1 (punto inicial): "))
        tol = 1e-8
        max_iter = 100
        xs_hist = [x0]
        for it in range(1, max_iter+1):
            fx = f(x0)
            d = deriv(f, x0)
            if not np.isfinite(fx) or not np.isfinite(d):
                print(f"Valores no finitos cerca de x={x0}. Deteniendo.")
                break
            if abs(d) < 1e-12:
                print(f"Derivada casi nula en x={x0:.6g}. Deteniendo.")
                break
            x1 = x0 - fx/d
            xs_hist.append(x1)
            if abs(x1 - x0) < tol:
                print(f"\nRaíz ≈ {x1:.10f} en {it} iteraciones")
                break
            x0 = x1
        else:
            print("\nNo convergió en el máximo de iteraciones.")

        try:
            plt.figure(figsize=(8,5))
            plt.plot(xs, ys, label="f(x)")
            plt.axhline(0, color="black", linestyle="--", linewidth=0.8)
            for xi in xs_hist:
                plt.plot([xi, xi], [0, f(xi)], alpha=.35)
                plt.scatter([xi], [f(xi)], s=28)
            plt.title("Iteraciones de Newton-Raphson")
            plt.legend(); plt.grid(True, alpha=.25)
            plt.show()
        except Exception:
            pass
    else:
        print("Operación cancelada.")
